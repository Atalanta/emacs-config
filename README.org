#+TITLE: SNS Emacs Config
#+AUTHOR: Stephen Nelson-Smith

This is my literate Emacs configuration. A single file holds both the prose explaining the config and the code that implements it.

All changes go in this file. Tangling produces the actual =.el= files Emacs loads.

* Instructions

Clone to =~/.emacs.d=:

#+BEGIN_EXAMPLE
cd $HOME
git clone git://github.com/Atalanta/emacs-config.git .emacs.d
#+END_EXAMPLE

Tangle the config:

#+begin_example
emacs --batch --load org --eval '(org-babel-tangle-file "README.org")'
#+end_example

First launch downloads and compiles packages, which takes a few minutes.

To change the config: edit this file, then run =org-babel-tangle= (=C-c C-v t=). Only blocks with a =:tangle= header are written out.

* Overview

Tangling =README.org= produces:

- =early-init.el=
- =init.el=
- =sns-emacs-modules/=
- =sns-lisp/=

* The ~early-init.el~ file

Emacs reads this file first. It runs before any packages load and sets a few UI defaults.

#+begin_src emacs-lisp :tangle "early-init.el"
  (setq inhibit-startup-message t)
  (tool-bar-mode -1)
  (scroll-bar-mode -1)
  (menu-bar-mode -1)
  (setq frame-inhibit-implied-resize t)
#+end_src

* The ~init.el~ file

Global settings and module imports.

** Support Nix
On machines where I use Nix, add the Nix profile to Emacs's =PATH=:

#+begin_src emacs-lisp :tangle "init.el"
(defun setup-nix-path ()
  "Add Nix profile to Emacs environment."
  (let ((nix-bin (expand-file-name ".nix-profile/bin" (getenv "HOME"))))
    (add-to-list 'exec-path nix-bin)
    (setenv "PATH" (concat nix-bin ":" (getenv "PATH")))))

(setup-nix-path)
#+end_src

** Ensure shell environment variables are available

GUI Emacs on macOS doesn't inherit shell environment variables. =exec-path-from-shell= imports them so Emacs can find tools like Git. Only run it under a window system; on a Linux terminal Emacs the environment is already correct.

#+begin_src emacs-lisp :tangle "init.el"
(use-package exec-path-from-shell
  :ensure t
  :config
  (when (memq window-system '(mac ns x))
    (exec-path-from-shell-initialize)))
#+end_src

** Use zsh as the shell

Use zsh for shell buffers and vterm. Fall back to bash where zsh isn't installed.

#+begin_src emacs-lisp :tangle "init.el"
  (setq shell-file-name (or (executable-find "zsh")
                             (executable-find "bash")
                             "/bin/bash"))
  (setq explicit-shell-file-name shell-file-name)
  (with-eval-after-load 'vterm
    (setq vterm-shell shell-file-name))
#+end_src

** Simplify auto-save and backup features
Git makes most of Emacs's backup features redundant. Disable backup files and lockfiles. Keep auto-save (Emacs's crash-recovery cache) but write it to a dedicated directory rather than alongside the working copy.

#+begin_src emacs-lisp :tangle "init.el"
(setq make-backup-files nil)

(setq auto-save-default t)
(setq auto-save-timeout 20)
(setq auto-save-interval 200)
(make-directory "~/.emacs.d/auto-save/" t)
(setq auto-save-file-name-transforms '((".*" "~/.emacs.d/auto-save/" t)))

(setq create-lockfiles nil)
#+end_src

** Allow short yes/no answers
Accept =y= and =n= instead of =yes= and =no=.

** Don't persist "customisations"
Emacs writes certain settings to =init.el= via its Customize system. All config in this repo is hand-written, so redirect those writes to a throwaway temp file.

** Run as a server

Run an Emacs server so external tools — =emacsclient=, scripts, AI agents
— can evaluate elisp in this Emacs. Pin the socket to =~/.emacs.d/server/=
so its path doesn't depend on =$TMPDIR= and is reachable from any shell.
Create the directory with mode =700= — =server.el= refuses to run if it's
group- or world-accessible, since a writable socket directory would let
other users spoof the server. Start the server only if one isn't already
running.

#+begin_src emacs-lisp :tangle "init.el"
  (require 'server)
  (setq server-socket-dir (expand-file-name "server/" user-emacs-directory))
  (with-file-modes #o700
    (make-directory server-socket-dir t))
  (unless (server-running-p)
    (server-start))
#+end_src

*** Org Roam

I use three commands daily, bound under =C-c r=:

- =C-c r f= - find or create a node
- =C-c r i= - insert a link to a node
- =C-c r c= - capture a new node

=org-roam-db-autosync-mode= keeps the link database current.

#+begin_src emacs-lisp :tangle "sns-emacs-modules/sns-knowledge-base.el"
  (use-package org-roam
    :ensure t
    :init
    (setq org-roam-v2-ack t)
    :custom
    (org-roam-directory sns-roam-directory)
    (org-roam-completion-everywhere t)
    :bind (("C-c r f" . org-roam-node-find)
           ("C-c r i" . org-roam-node-insert)
           ("C-c r c" . org-roam-capture))
    :config
    (org-roam-db-autosync-mode))

  (provide 'sns-knowledge-base)

  "Minor mode for highlighting double spaces using theme-defined colors."
  :lighter " DSpaces"
  :global nil
  :init-value nil
  (if highlight-double-spaces-mode
      (progn
        (font-lock-add-keywords nil '(("  +" 0 'highlight t)))
        (font-lock-flush)
        (font-lock-ensure))

*** Rename file and buffer
#+begin_src emacs-lisp :tangle "sns-lisp/sns-functions.el" :mkdirp yes
  (defun sns-rename-file-and-buffer ()
    "Rename the file visited by the current buffer and update the buffer to visit the renamed file.
  Prompts for a new name and handles version-controlled files appropriately. If the buffer is not
  visiting a file, signals an error. Ensures the buffer is updated after renaming."
    (interactive)
    (let ((filename (buffer-file-name)))
      (unless (and filename (file-exists-p filename))
	(user-error "Buffer is not visiting a file"))
      (let ((new-name (read-file-name "New name: " filename)))
	(if (vc-backend filename)
	    ;; Use version control rename if the file is under version control.
	    (vc-rename-file filename new-name)
	  ;; Rename file normally and update buffer.
	  (progn
	    (rename-file filename new-name t)
	    (set-visited-file-name new-name t t))))))
#+end_src

*** Reader view for org mode

A toggle for prose-friendly display: wider margins, extra line spacing, no line numbers. Useful when writing in the journal.

#+begin_src emacs-lisp :tangle "sns-lisp/sns-functions.el" :mkdirp yes
(use-package visual-fill-column
  :ensure t
  :config
  (setq-default visual-fill-column-center-text t)

  (defun sns-toggle-readable-view ()
    "Toggle a 'readable' view with centered text, wider margins, and hidden line numbers."
    (interactive)
    (if (bound-and-true-p visual-fill-column-mode)
        (progn
          (visual-fill-column-mode -1)
          (setq line-spacing nil)
          (display-line-numbers-mode 1)
          (message "Readable view disabled"))
      (progn
        (setq visual-fill-column-width 80)
        (setq visual-fill-column-center-text t)
        (setq line-spacing 0.3)
        (display-line-numbers-mode -1)
        (visual-fill-column-mode 1)
        (message "Readable view enabled")))))
#+end_src
